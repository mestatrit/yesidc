$("#submit").bind("click", function() {
    if (!this.ajaxing) {
        this.ajaxing = true;
        this.innerHTML = '提交订单中<span class="ani_dot">...</span>';
        setTimeout(function() {
            this.ajaxing = false;
            this.innerHTML = "提交超时";
        }.bind(this), 30000);
    }
});
